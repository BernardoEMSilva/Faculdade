def NumeroDeNegativos (Array):
    Contador = 0
    for i in range(3):
        for j in range(4):
            if Array[i][j]<0:
                Contador += 1
    return Contador

array = [ [10, 20, -10, -20], [40, 50, -30, 30], [60, 70, -60, -70]]

print('No Array existem ',NumeroDeNegativos(array),'numeros negativos')
