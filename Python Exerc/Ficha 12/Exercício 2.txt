def Escrever(Frase):
    try:
        fo = open('texto.txt', 'w')
        fo.write(Frase)
        fo.close()
    except:
        print('Problema na abertuta do ficheiro e escrita do ficheiro, encerar programa')

def Lerfich():
    try:
        fo = open('texto.txt', 'r')
        ch = fo.read()
        ch = ch.split()
        fo.close()
        return ch
    except:
        print('Problema na abertuta do ficheiro e leitura do ficheiro, encerar programa')
